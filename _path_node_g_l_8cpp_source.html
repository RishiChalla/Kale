32void PathNode::setup() {

69void PathNode::updateBoundingBox() {
70 Collidable::boundingBox = path.getBoundingBox();
71
72 if (stroke == StrokeStyle::Outside || stroke == StrokeStyle::Both) {
73 Collidable::boundingBox.topLeft -= strokeRadius;
74 Collidable::boundingBox.bottomRight += strokeRadius;
75 }
76
77 const std::array<Vector2f, 4> verts = {
78 Collidable::boundingBox.bottomLeft(),
79 Collidable::boundingBox.topLeft,
80 Collidable::boundingBox.bottomRight,
81 Collidable::boundingBox.topRight()
82 };
83
84 std::copy(reinterpret_cast<const float*>(verts.cbegin()), reinterpret_cast<const float*>(verts.end()), vertexArray->vertices.data.begin());
85
86 // OpenGL commands must be run on the main thread - add a task for it.
87 mainApp->runTaskOnMainThread([&]() { vertexArray->vertices.updateBuffer(); });
88}

123void PathNode::preUpdate(size_t threadNum, const Scene& scene, float deltaTime) {
124 // Call transformable update
125 Transformable::updateTransform(deltaTime);
126
127 // Update the Path based on the FSM if applicable
128 if (pathFSM.has_value()) {
129
130 // Update the state
131 pathFSM->updateState(deltaTime);
132
133 // Clear the path
134 std::fill(path.beziers.begin(), path.beziers.end(), CubicBezier{Vector2f::zero(), Vector2f::zero(), Vector2f::zero(), Vector2f::zero()});
135
136 // Loop through the composition & lerp between states as applicable
137 for (std::pair<int, float> composition : pathFSM->getStateComposition<int>()) {
138 path += pathFSM->getStructure<int>(composition.first) * composition.second;
139 }
140
141 // Update the bounding box
142 updateBoundingBox();
143 }
144}

150void PathNode::render(const Camera& camera, float deltaTime) const {
151 // There is no vertex array setup - nothing to render
152 if (vertexArray == nullptr) return;
153
154 // Use the shader & provide uniforms
155 shader->useProgram();
156 shader->uniform(cameraUniform, camera);
157 shader->uniform(localUniform, getFullTransform());
158 shader->uniform(vertexColorUniform, color);
159 shader->uniform(strokeColorUniform, strokeColor);
160 shader->uniform(zPosition, zPosition);
161 shader->uniform(fillUniform, fill ? 1 : 0);
162 shader->uniform(strokeUniform, static_cast<int>(stroke));
163 shader->uniform(strokeRadiusUniform, strokeRadius);
164
165 shader->uniform(beziersUniform, reinterpret_cast<const Vector2f*>(path.beziers.data()), path.beziers.size() * 4);
166 shader->uniform(numBeziersUniform, static_cast<int>(path.beziers.size()));
167
168 // Draw, fragment shaders will do the rest of the work for us
169 vertexArray->draw();
170}
