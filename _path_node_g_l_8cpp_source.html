71void PathNode::begin(const Scene& scene) {
72 Rect boundingBox = path.getBoundingBox();
73 if (stroke != StrokeStyle::Neither) {
74 boundingBox.topLeft -= strokeRadius;
75 boundingBox.bottomRight += strokeRadius;
76 }
77
78 const std::array<Vector2f, 4> verts = {boundingBox.bottomLeft(), boundingBox.topLeft, boundingBox.bottomRight, boundingBox.topRight()};
79 const std::array<unsigned int, 6> indices = {0, 1, 2, 1, 3, 2};
80
81 vertexArray = std::make_unique<OpenGL::VertexArray<Vector2f, 2>>(verts, indices, OpenGL::BufferUsage::Static);
82 vertexArray->enableAttributePointer({posAttribute});
83}

89void PathNode::render(const Camera& camera, float deltaTime) const {
90 if (vertexArray == nullptr) return;
91 shader->useProgram();
92 shader->uniform(cameraUniform, camera);
93 shader->uniform(localUniform, transform);
94 shader->uniform(vertexColorUniform, color);
95 shader->uniform(strokeColorUniform, strokeColor);
96 shader->uniform(zPosition, zPosition);
97 shader->uniform(fillUniform, fill ? 1 : 0);
98 shader->uniform(strokeUniform, static_cast<int>(stroke));
99 shader->uniform(strokeRadiusUniform, strokeRadius);
100
101 shader->uniform(beziersUniform, reinterpret_cast<const Vector2f*>(path.beziers.data()), path.beziers.size() * 4);
102 shader->uniform(numBeziersUniform, static_cast<int>(path.beziers.size()));
103
104 vertexArray->draw();
105}
