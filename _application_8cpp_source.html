116void Application::synchronizeUpdate() {
117 std::unique_lock lock(threadSyncMutex);
118 numThreadsUpdated--;
119 renderingFinished = false;
120
121 if (numThreadsUpdated == 0) {
122 updatingFinished = true;
123 numThreadsUpdated = getNumUpdateThreads();
124 renderSyncCondVar.notify_one();
125 }
126
127 threadSyncCondVar.wait(lock, [&]() -> bool { return renderingFinished; });
128}

134void Application::update(size_t threadNum) noexcept {
135
136 // Update loop
137 while (window.isOpen()) {
138
139 // Wait until we should update
140 synchronizeUpdate();
141
142 // Perform updating
143 if (presentedScene != nullptr) try {
144 presentedScene->update(threadNum, deltaTime);
145 }
146 catch (const std::exception& e) {
147 console.error("Failed to update presented screen on update thread " + std::to_string(threadNum) + " - " + e.what());
148 }
149 }
150
151 // Avoid the main thread waiting for update threads which are already finished
152 synchronizeUpdate();
153}

158void Application::run() noexcept {
159
160 using namespace std::string_literals;
161 using namespace std::chrono_literals;
162
163 // Creates the window
164 window.create(applicationName.c_str());
165
166
167#ifdef KALE_VULKAN
168 // Setup Vulkan
169 Vulkan::Core::setupCore();
170#endif
171
172#ifdef KALE_OPENGL
173 // Setup OpenGL
174 OpenGL::Core::setupCore();
175#endif
176
177 try {
178 onBegin();
179 }
180 catch (const std::exception& e) {
181 console.error("Failed to call onBegin in application - "s + e.what());
182 }
183
184 // Setup nodes
185 try {
186 PathNode::setup();
187 }
188 catch (const std::exception& e) {
189 console.error("Terminating application due to failure to setup nodes - "s + e.what());
190 return;
191 }
192
193 // Create update threads
194 numThreadsUpdated = std::thread::hardware_concurrency();
195 for (size_t i = 0; i < std::thread::hardware_concurrency(); i++)
196 updateThreads.emplace_back(&Application::update, this, i);
197
198 // Render loop
199 auto previousTime = std::chrono::high_resolution_clock::now();
200 while (window.isOpen()) {
201
202 // Update the window for event polling, etc
203 window.update();
204
205 // Calculate FPS
206 auto currentTime = std::chrono::high_resolution_clock::now();
207 deltaTime = static_cast<float>(std::chrono::duration_cast<std::chrono::microseconds>(currentTime - previousTime).count());
208 previousTime = std::chrono::high_resolution_clock::now();
209
210 // Check if scene has changed prior to updating
211 if (sceneToPresent != nullptr) {
212 try {
213 sceneToPresent->onPresent();
214 if (presentedScene != nullptr) presentedScene->onSceneChange();
215 presentedScene = sceneToPresent;
216 sceneToPresent = nullptr;
217 }
218 catch (const std::exception& e) {
219 console.error("Failed to present scene - "s + e.what());
220 }
221 }
222
223 // Synchronize with the update threads
224 {
225 std::unique_lock lock(threadSyncMutex);
226 updatingFinished = false;
227 renderingFinished = true;
228 threadSyncCondVar.notify_all();
229 renderSyncCondVar.wait(lock, [&]() -> bool { return updatingFinished; });
230 }
231
232 // Run all tasks required
233 while (!tasks.empty()) try {
234 tasks.front()();
235 tasks.pop();
236 }
237 catch (const std::exception& e) {
238 console.error("Failed to execute task on main thread - "s + e.what());
239 }
240
241 if (presentedScene != nullptr) try {
242 // Update node structures
243 presentedScene->updateNodeStructures();
244 // Render scene
245 presentedScene->render(deltaTime);
246 }
247 catch (const std::exception& e) {
248 console.error("Failed to render presented scene - "s + e.what());
249 }
250 }
251
252 // Wait for threads
253 renderingFinished = true;
254 threadSyncCondVar.notify_all();
255 for (std::thread& thread : updateThreads) thread.join();
256
257 onEnd();
258
259 // Cleanup nodes
260 try {
261 PathNode::cleanup();
262 }
263 catch (const std::exception& e) {
264 console.error("Terminating application due to failure to cleanup nodes - "s + e.what());
265 return;
266 }
267
268#ifdef KALE_VULKAN
269 // Cleanup vulkan now that execution is done
270 Vulkan::Core::cleanupCore();
271#endif
272
273#ifdef KALE_OPENGL
274 // Setup OpenGL
275 OpenGL::Core::cleanupCore();
276#endif
277}
