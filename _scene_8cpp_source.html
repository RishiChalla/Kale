73void Scene::updateNodeStructures() {
74
75 // Return if no nodes need to be added
76 if (nodesToAdd.empty() && nodesToRemove.empty()) return;
77
78 // Add all the nodes till there's none left to add
79 while (!nodesToAdd.empty()) {
80 // Get the node and add it to the main list
81 std::shared_ptr<Node> node = nodesToAdd.front();
82 nodesToAdd.pop();
83 nodes.push_back(node);
84
85 // Find the thread with the current smallest total update time
86 size_t threadIndex = std::distance(threadedNodePerformanceTimes.begin(),
87 std::min_element(threadedNodePerformanceTimes.begin(), threadedNodePerformanceTimes.end(),
88 [](const std::pair<float, float>& a, const std::pair<float, float>& b) -> bool {
89 return a.first > b.first;
90 }
91 )
92 );
93
94 // Add the node to the thread with the smallest update time and add it to the thread's total time
95 threadedNodePerformanceTimes[threadIndex].first += node->updateTime;
96 updateNodes[threadIndex].push_back(node);
97
98 // Find the thread with the current smallest total pre update time
99 threadIndex = std::distance(threadedNodePerformanceTimes.begin(),
100 std::min_element(threadedNodePerformanceTimes.begin(), threadedNodePerformanceTimes.end(),
101 [](const std::pair<float, float>& a, const std::pair<float, float>& b) -> bool {
102 return a.second > b.second;
103 }
104 )
105 );
106
107 // Add the node to the thread with the smallest pre update time and add it to the thread's total time
108 threadedNodePerformanceTimes[threadIndex].second += node->preUpdateTime;
109 preUpdateNodes[threadIndex].push_back(node);
110
111 node->begin(*this);
112 }
113
114 while (!nodesToRemove.empty()) {
115 // Get the node and remove it from the main list
116 std::shared_ptr<Node> node = nodesToRemove.front();
117 nodesToRemove.pop();
118 nodes.remove(node);
119
120 bool updateFound = false;
121 bool preUpdateFound = false;
122
123 // Loop through the threads till we find the thread holding the update
124 for (size_t threadIndex = 0; threadIndex < updateNodes.size(); threadIndex++) {
125
126 if (updateFound && preUpdateFound) break;
127
128 if (!updateFound) {
129 // Check if the node is updated in this thread
130 auto it = std::find(updateNodes[threadIndex].begin(), updateNodes[threadIndex].end(), node);
131 if (it != updateNodes[threadIndex].end()) {
132 // Remove the node from updates & update the performance times
133 threadedNodePerformanceTimes[threadIndex].first -= node->updateTime;
134 updateNodes[threadIndex].erase(it);
135 updateFound = true;
136 }
137 }
138
139 if (!preUpdateFound) {
140 // Check if the node is pre updated in this thread
141 auto it = std::find(preUpdateNodes[threadIndex].begin(), preUpdateNodes[threadIndex].end(), node);
142 if (it != preUpdateNodes[threadIndex].end()) {
143 // Remove the node from updates & update the performance times
144 threadedNodePerformanceTimes[threadIndex].second -= node->preUpdateTime;
145 preUpdateNodes[threadIndex].erase(it);
146 preUpdateFound = true;
147 }
148 }
149 }
150
151 node->end(*this);
152 }
153}

184void Scene::update(size_t threadNum, float deltaTime) {
185
186 // Pre updating
187 onPreUpdate(threadNum, deltaTime);
188 for (std::shared_ptr<Node>& node : preUpdateNodes[threadNum]) node->preUpdate(threadNum, *this, deltaTime);
189
190 // mark pre updating as done & notify other threads if necessary
191 {
192 std::unique_lock lock(nodePreUpdateMutex);
193 nodesPreUpdated--;
194 size_t localGeneration = generation;
195 if (nodesPreUpdated == 0) {
196 generation++;
197 nodesPreUpdated = mainApp->getNumUpdateThreads();
198 nodePreUpdateCondVar.notify_all();
199 }
200 else {
201 nodePreUpdateCondVar.wait(lock, [&]() -> bool { return localGeneration != generation; });
202 }
203 }
204
205 // notify other threads if applicable
206 if (nodesPreUpdated == mainApp->getNumUpdateThreads()) nodePreUpdateCondVar.notify_all();
207
208 // Updating
209 onUpdate(threadNum, deltaTime);
210 for (std::shared_ptr<Node>& node : updateNodes[threadNum]) node->preUpdate(threadNum, *this, deltaTime);
211}
