278 template <size_t w2> Matrix<w2, h, T> operator*(const Matrix<w2, w, T>& other) const {
279 Matrix<w2, h, T> mat;
280 for (size_t i = 0; i < h; i++)
281 for (size_t j = 0; j < w2; j++)
282 for (size_t k = 0; k < w; k++)
283 mat.get(j, i) += get(k, i) * other.get(j, k);

307 void swapRows(size_t row1, size_t row2) {
308 #ifdef KALE_DEBUG
309 if (row1 > h || row2 > h) throw std::runtime_error("Invalid matrix row");

311 std::swap_ranges(data.begin() + row1 * w, data.begin() + row1 * w + w, data.begin() + row2 * w);

336 void addScaledRow(size_t row1, size_t row2, T scalar) {

339 #endif
340 for (size_t x = 0; x < w; x++) {
341 get(x, row1) += scalar * get(x, row2);
342 }
