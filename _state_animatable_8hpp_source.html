85 void updateState(float deltaTime) {
86 if (!transitioning) return;
87 transitionTime += deltaTime / (1000000.0f * transitionDuration);
88
89 // Animation complete
90 if (transitionTime >= 1.0f) {
91 state = transitionState;
92
93 // There is no further animation required
94 if (animationInfo.empty()) {
95 transitioning = false;
96 return;
97 }
98
99 // We need to move to the next animation stage
100 if (++animationIndex >= animationInfo.size()) {
101 // We don't need to loop, we've finished.
102 if (!animationLoop) {
103 animationInfo.clear();
104 transitioning = false;
105 return;
106 }
107
108 // We finished going through the animation info, check if we need to repeat
109 animationIndex = 0;
110 }
111
112 // Prepare the next animation
113 transitionTime = 0.0f;
114 transitionDuration = animationInfo[animationIndex].second;
115 transitionState = animationInfo[animationIndex].first;
116 }
117 }
