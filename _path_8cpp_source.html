52Rect Path::getBoundingBox() const {
53 Vector2f topLeft = Vector2f::max();
54 Vector2f bottomRight = Vector2f::min();
55
56 for (const CubicBezier& bezier : beziers) {
57 for (const Vector2f& point : {bezier.start, bezier.controlPoint1, bezier.controlPoint2, bezier.end}) {
58 if (point.x < topLeft.x) topLeft.x = point.x;
59 if (point.y < topLeft.y) topLeft.y = point.y;
60 if (point.x > bottomRight.x) bottomRight.x = point.x;
61 if (point.y > bottomRight.y) bottomRight.y = point.y;
62 }
63 }
64
65 return {topLeft, bottomRight};
66}
