220void Transform::setRotation(float angle, AngleUnit unit) {
221 if (unit == AngleUnit::Degree) angle = degToRad(angle);
222 Vector2f scale = getScale();
223 float c = cos(angle);
224 float s = sin(angle);
225 data[0] = scale.x * c;
226 data[1] = scale.y * -s;
227 data[3] = scale.x * s;
228 data[4] = scale.y * c;
229}

298Vector2f Transform::inverseTransform(const Vector2f& vec) const {
299 float den = data[0] * data[4] - data[1] * data[3];
300 return {
301 (data[1] * data[5] - data[1] * vec.y - data[2] * data[4] + data[4] * vec.x) / den,
302 (-data[0] * data[5] + data[0] * vec.y + data[2] * data[3] - data[3] * vec.x) / den
303 };
304 // Below system of equations represents transformation where k, j = vec.x, vec.y
305 // x * n0 + y * n1 + n2 = k
306 // x * n3 + y * n4 + n5 = j
307 // Solve the above for x & y ->
308 // x = (n1*n5 - n1*j - n2*n4 + n4*k) / (n0*n4 - n1*n3)
309 // y = (-n0*n5 + n0*j + n2*n3 - n3*k) / (n0*n4 - n1*n3)
310}

317void Transform::inverseTransformInplace(Vector2f& vec) const {
318 float den = data[0] * data[4] - data[1] * data[3];
319 vec.x = (data[1] * data[5] - data[1] * vec.y - data[2] * data[4] + data[4] * vec.x) / den;
320 vec.y = (-data[0] * data[5] + data[0] * vec.y + data[2] * data[3] - data[3] * vec.x) / den;
321}

386Ray Transform::transform(const Ray& ray) const {
387 Ray newRay;
388 newRay.origin = transform(ray.origin);
389 newRay.direction = transform(ray.direction + ray.origin) - newRay.origin;
390 newRay.direction /= newRay.direction.magnitude();
391 return newRay;
392}
