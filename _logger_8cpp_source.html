45void Logger::load(const std::string& applicationName) {
46
47 // Create the folder for this application if it doesn't already exist
48 if (!std::filesystem::exists("." + applicationName))
49 std::filesystem::create_directory("." + applicationName);
50 if (!std::filesystem::exists("." + applicationName + "/logs"))
51 std::filesystem::create_directory("." + applicationName + "/logs");
52
53 // Create/open the log file in the correct folder
54 logFile.open("." + applicationName + "/logs/" + date::format("%F--%H-%M", std::chrono::system_clock::now()) + ".log");
55
56 #ifdef KALE_WINDOWS
57 HANDLE hOut = GetStdHandle(STD_OUTPUT_HANDLE);
58 if (hOut == INVALID_HANDLE_VALUE) {
59 return;
60 }
61
62 DWORD dwMode = 0;
63 if (!GetConsoleMode(hOut, &dwMode)) {
64 return;
65 }
66
67 dwMode |= ENABLE_VIRTUAL_TERMINAL_PROCESSING;
68 if (!SetConsoleMode(hOut, dwMode)) {
69 return;
70 }
71 #endif
72}
