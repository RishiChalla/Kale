25void Transformable::updateTransform(float deltaTime) {
26 // Update the Transform based on the FSM if applicable
27 if (!transformFSM.has_value()) return;
28
29 // Update the state
30 transformFSM->updateState(deltaTime);

32 // Clear the transform

36 0.0f, 0.0f, 0.0f
37 });
38
39 // Loop through the composition & lerp between states as applicable
40 for (std::pair<int, float> composition : transformFSM->getStateComposition<int>())
41 *stateTransform += transformFSM->getStructure<int>(composition.first) * composition.second;
