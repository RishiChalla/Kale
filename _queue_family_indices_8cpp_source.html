37QueueFamilyIndices::QueueFamilyIndices(const vk::PhysicalDevice& device) {
38
39 // Loop through the family properties and set each index
40 uint32_t i = 0;
41 for (const vk::QueueFamilyProperties& familyProperties : device.getQueueFamilyProperties()) {
42 // Check the queue flags and set the indices based on them
43 if (familyProperties.queueFlags & vk::QueueFlagBits::eGraphics) graphicsFamilyIndex = i;
44 if (device.getSurfaceSupportKHR(i, Core::surface.get())) presentFamilyIndex = i;
45
46 // Break the loop if we've completed all indices
47 if (hasAllIndices()) break;
48 i++;
49 }
50}
